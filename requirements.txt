# ─────────────────────────────────────────────
# Runtime dependencies — what the app needs to run
# ─────────────────────────────────────────────

# Phase 1: PDF text extraction
# PyMuPDF gives us coordinate-aware PDF text extraction.
# We need version 1.23+ because earlier versions had breaking API changes
# in get_text("dict") that we rely on for block-level bounding boxes.
pymupdf>=1.23.0

# Phase 3: ML niche classifier
# scikit-learn provides the RandomForestClassifier and evaluation metrics.
scikit-learn>=1.4

# pandas loads the labeled training dataset CSV and JSON files.
pandas>=2.0

# numpy is required by scikit-learn and for feature vector array operations.
numpy>=1.24

# joblib serializes and deserializes the trained model artifact (.joblib files).
joblib>=1.3

# pyyaml reads the training configuration YAML file (config/training_config.yaml).
pyyaml>=6.0

# Phase 5: radar chart generation (in-memory PNG via Agg backend — no display needed).
matplotlib>=3.8

# Phase 5: PDF generation using PLATYPUS layout engine (entirely in-memory, no disk writes).
reportlab>=4.0

# Phase 6: Web frontend — Flask serves the upload form and JSON API.
# Flask 3.0+ is required because it added async support and tightened the
# application-factory pattern that create_app() relies on.
Flask>=3.0

# waitress is a production-grade WSGI server for Windows.
# It replaces Flask's single-threaded dev server so concurrent visitors
# don't queue behind each other during an ML analysis.
waitress>=3.0
